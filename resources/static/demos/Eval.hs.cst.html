 78: ------------------------------------------------------------------
 79: -- The set of binders in an environment --------------------------
 80: ------------------------------------------------------------------
 81: 
 82: {-@ measure vars :: Env -> (Set Bndr)
 83:     vars ([])    = {v | (Set_emp v) }
 84:     vars (b:env) = {v | v = (Set_cup (Set_sng (fst b)) (vars env))}
 85:   @-}

 87: ------------------------------------------------------------------
 88: -- The set of free-vars in an expression -------------------------
 89: ------------------------------------------------------------------
 90: 
 91: {-@ measure free       :: Expr -> (Set Bndr) 
 92:     free (Const i)     = {v | (Set_emp v)}
 93:     free (Var x)       = {v | v = (Set_sng x)} 
 94:     free (Plus e1 e2)  = {v | v = (Set_cup (free e1) (free e2))}
 95:     free (Let x e1 e2) = {v | v = (Set_cup (free e1) (Set_dif (free e2) (Set_sng x)))}
 96:     free (App e1 e2)   = {v | v = (Set_cup (free e1) (free e2))}
 97:     free (Lam x e)     = {v | v = (Set_dif (free e) (Set_sng x))}
 98:   @-}

101: -- A Size Metric on Expressions ----------------------------------
102: ------------------------------------------------------------------
103: 
104: {-@ measure esize       :: Expr -> Int 
105:     esize (Var x)       = 1
106:     esize (Const i)     = 1
107:     esize (Plus e1 e2)  = (esize e1) + (esize e2)
108:     esize (Let x e1 e2) = (esize e1) + (esize e2)
