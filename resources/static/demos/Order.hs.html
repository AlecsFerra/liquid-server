 64: -- Dependent Tuples
 65: -- ----------------
 66: 
 67: -- `break` from the Prelude.
 68: 
 69: break                   :: (a -> Bool) -> [a] -> ([a], [a])
 70: break _ xs@[]           =  (xs, xs)
 71: break p xs@(x:xs')
 72:            | p x        =  ([], xs)
 73:            | otherwise  =  let (ys, zs) = break p xs'
 74:                            in (x:ys,zs)

 76: -- Dependent Tuples via Abstract Refinements
 77: -- 
 78: -- data (a,b)<p :: a -> b -> Prop> = (x:a, b<p x>)
 79: 
 80: -- Instantiate the `p` in *different* ways.
 81: 
 82: {-@ plusOnes :: [(Int, Int)<{\x1 x2 -> x2 = x1 + 1}>] @-}
 83: plusOnes = [(0,1), (5,6), (999,1000)]

 86:           -> ([a], [a])<{\y z -> (Break x y z)}> @-}

 88: {-@ predicate Break X Y Z   = (len X) = (len Y) + (len Z) @-}

 90: -- Abstractly Refined Lists
 91: -- ------------------------
 92: 
 93: -- data [a] <p :: a -> a -> Prop> where
 94: --   | []  :: [a] <p>
 95: --   | (:) :: h:a -> [a<p h>]<p> -> [a]<p>
 96: 
 97: -- * The type is parameterized with a refinement `p :: a -> a -> Prop`
 98: --   Think of `p` as a *binary relation* over the `a` values comprising
 99: --   the list.
100: 
101: -- * The empty list `[]` is a `[]<p>`. Clearly, the empty list has no
102: --   elements whatsoever and so every pair is trivially, or rather,
103: --   vacuously related by `p`.
104: 
105: -- * The cons constructor `(:)` takes a head `h` of type `a` and a tail
106: --   of `a<p h>` values, each of which is *related to* `h` **and** which
107: --   (recursively) are pairwise related `[...]<p>` and returns a list where
108: --   *all* elements are pairwise related `[a]<p>`.
109: 
110: -- Using Abstractly Refined Lists
111: -- ------------------------------
112: 
113: -- For starters, we can define a few helpful type aliases.
114: 
115: {-@ type IncrList a = [a]<{\xi xj -> xi <= xj}> @-}
116: {-@ type DecrList a = [a]<{\xi xj -> xi >= xj}> @-}
117: {-@ type UniqList a = [a]<{\xi xj -> xi /= xj}> @-}
118: 
119: 
120: {-@ whatGosUp :: IncrList Integer @-}
121: whatGosUp = [1,2,3]
122: 
123: 
124: {-@ mustGoDown :: DecrList Integer @-}
125: mustGoDown = [3,2,1]
126: 
127: 
128: {-@ noDuplicates :: UniqList Integer @-}
129: noDuplicates = [1,3,2]

132: -- Sorting Lists
133: -- -------------
134: 
135: -- **Insertion Sort**
136: 
137: {-@ insertSort    :: (Ord a) => xs:[a] -> (IncrList a) @-}
138: insertSort []     = []
139: insertSort (x:xs) = insert x (insertSort xs)
140: 
141: {-@ insert :: (Ord a) => a -> IncrList a -> IncrList a @-}
142: insert y []     = [y]
143: insert y (x:xs)
144:   | y <= x      = y : x : xs
145:   | otherwise   = x : insert y xs
146: 
147: 
148: -- If you prefer the more Haskelly way of writing insertion sort,
149: -- i.e. with a `foldr`, that works too. Can you figure out why?
150: 
151: {-@ insertSort' :: (Ord a) => [a] -> IncrList a @-}
152: insertSort' xs  = foldr insert [] xs
153: 
154: 
155: -- **Merge Sort**
156: 
157: 
158: split          :: [a] -> ([a], [a])
159: split (x:y:zs) = (x:xs, y:ys)
160:   where
161:     (xs, ys)   = split zs
162: split xs       = (xs, [])
163: 
164: 
165: {-@ merge :: (Ord a) => IncrList a -> IncrList a -> IncrList a @-}
166: merge xs []         = xs
167: merge [] ys         = ys
168: merge (x:xs) (y:ys)
169:   | x <= y          = x : merge xs (y:ys)
170:   | otherwise       = y : merge (x:xs) ys
171: 
172: {-@ mergeSort :: (Ord a) => [a] -> IncrList a @-}
173: mergeSort []  = []
174: mergeSort [x] = [x]
175: mergeSort xs  = merge (mergeSort ys) (mergeSort zs)
176:   where
177:     (ys, zs)  = split xs
178: 
179: 
180: -- **Quick Sort**
181: 
182: 
183: {-@ quickSort    :: (Ord a) => [a] -> IncrList a @-}
184: quickSort []     = []
185: quickSort (x:xs) = pivApp x lts gts
186:   where
187:     lts          = quickSort [y | y <- xs, y < x ]
188:     gts          = quickSort [z | z <- xs, z >= x]
189: 
190: pivApp piv []     ys  = piv : ys
191: pivApp piv (x:xs) ys  = x   : pivApp piv xs ys

194: -- Really Sorting Lists
195: -- --------------------
196: 
197: -- The convenient thing about our encoding is that the
198: -- underlying datatype is plain Haskell lists.
199: -- By decoupling (or rather, parameterizing)
200: -- the relation or property or invariant from the actual
201: -- data structure we can plug in different invariants,
202: -- sometimes in the *same* program.
203: 
204: -- To see why this is useful, lets look at a *real-world*
205: -- sorting algorithm: the one used inside GHC's
206: -- `Data.List` [module][ghc-list].
207: 
208: {-@ sort :: (Ord a) => [a] -> IncrList a  @-}
209: sort = mergeAll . sequences
210:   where
211:     sequences (a:b:xs)
212:       | a `compare` b == GT = descending b [a]  xs
213:       | otherwise           = ascending  b (a:) xs
214:     sequences [x] = [[x]]
215:     sequences []  = [[]]
216: 
217:     descending a as (b:bs)
218:       | a `compare` b == GT = descending b (a:as) bs
219:     descending a as bs      = (a:as): sequences bs
220: 
221:     ascending a as (b:bs)
222:       | a `compare` b /= GT = ascending b (\ys -> as (a:ys)) bs
223:     ascending a as bs       = as [a]: sequences bs
224: 
225:     mergeAll [x] = x
226:     mergeAll xs  = mergeAll (mergePairs xs)
227: 
228:     mergePairs (a:b:xs) = merge a b: mergePairs xs
229:     mergePairs [x]      = [x]
230:     mergePairs []       = []
